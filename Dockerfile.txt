
# Estágio 1: Builder
FROM node:22-alpine AS builder

WORKDIR /app

# Necessário para algumas dependências nativas se houver
RUN apk add --no-cache python3 make g++

COPY package*.json ./
RUN npm install

# Copia todo o código-fonte
COPY . .

# Executa build do Vite e do TSC (configurado no package.json)
RUN npm run build

# Estágio 2: Runner
FROM node:22-alpine

WORKDIR /app

# Copia apenas os arquivos necessários do estágio anterior
COPY package*.json ./
RUN npm install --omit=dev

# Copia os artefatos compilados (frontend e backend)
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/dist_server ./dist_server

# Expõe a porta do core
EXPOSE 3005

# Ambiente de produção
ENV NODE_ENV=production

# Comando de inicialização
CMD ["node", "dist_server/server.js"]
